from IPython.display import clear_output

def do_experiment(N, T, cutoff=0.05):
    df = pd.DataFrame()

    # Make random data
    for i in range(N):
        X = np.random.normal(0, 1, T)
        X = pd.Series(X)
        name = 'X%s' % i
        df[name] = X

    significant_pairs = []

    # Look for relationships
    for i in range(N):
        for j in range(i+1, N):
            Xi = df.iloc[:, i]
            Xj = df.iloc[:, j]

            results = stats.spearmanr(Xi, Xj)

            pvalue = results[1]

            if pvalue < cutoff:
                significant_pairs.append((i, j))

    return significant_pairs


num_experiments = 100

results = np.zeros((num_experiments,))

for i in range(num_experiments):
    print(f'trial {i+1} of {num_experiments}')
    clear_output(wait=True)

    # Run a single experiment
    result = do_experiment(20, 100, cutoff=0.05)

    # Count how many pairs
    n = len(result)

    # Add to array
    results[i] = n

print('The average result is:', np.mean(results))

def get_pvalues_from_experiment(N, T):
    df = pd.DataFrame()

    # Make random data
    for i in range(N):
        X = np.random.normal(0, 1, T)
        X = pd.Series(X)
        name = 'X%s' % i
        df[name] = X

    pvalues = []

    # Look for relationships
    for i in range(N):
        for j in range(i+1, N):
            Xi = df.iloc[:, i]
            Xj = df.iloc[:, j]

            results = stats.spearmanr(Xi, Xj)

            pvalue = results[1]

            pvalues.append(pvalue)

    return pvalues

Gim Thiu Thin Kin Do So Snh Nhiu

Bn khng th thc s loi b thin kin do so snh nhiu, nhng bn c th gim thiu nh hng ca n. lm iu , chng ta c hai ty chn.

Ty chn 1: Thc hin t th nghim hn.

iu ny thng l la chn tt nht. Thay v ch qut xung quanh hy vng bn tm thy mt tn hiu th v, hy s dng kin thc chuyn mn ca bn v h thng pht trin mt gi thuyt tt v kim tra n. Quy trnh khm ph d liu, a ra gi thuyt, sau thu thp thm d liu v kim tra gi thuyt trn d liu mi c coi l tiu chun vng trong nghin cu thng k v khoa hc. iu quan trng l b d liu m bn pht trin gi thuyt khng phi l b d liu m bn kim tra. Bi v bn tm thy hiu ng trong khi khm ph, bi kim tra s qua v khng thc s cho bn bit iu g. iu bn mun bit l hiu ng ny c nht qun khng. Chuyn sang d liu mi v kim tra khng ch c ngha l bn ch thc hin mt th nghim, m cn l mt 'c lng khng thin lch' v vic gi thuyt ca bn c ng hay khng.

Ty chn 2: Cc yu t iu chnh v Sa i Bon Ferroni

CNH BO: Phn ny c th hi k thut. Tr khi bn cm thy thoi mi vi cc mc ngha, chng ti khuyn bn nn xem xt cc v d code trc v c th c cc bi vit lin kt trc khi hon ton tip cn on ny.

Nu bn phi thc hin nhiu th nghim, hy c gng iu chnh cc p-value ca bn. iu ny c ngha l p dng mt yu t iu chnh cho ngng bn mong mun c c ngng thc t c s dng khi xc nh xem cc p-value c ngha hay khng. Yu t iu chnh bo th v ph bin nht l Bon Ferroni.

V d: Sa i Bon Ferroni

Khi nim ng sau Bon Ferroni rt n gin. N ch ra rng nu chng ta thc hin $m$ th nghim, v chng ta c mc ngha/cutoff l $a$, th chng ta nn s dng $a/m$ lm cutoff mi ca mnh khi xc nh ngha. Ton hc din ra nh sau.

Gi s chng ta thc hin $m$ th nghim. Chng ta nn mong i thy $ma$ kt qu dng gi da trn c hi ngu nhin vt qua cutoff ca chng ta. Nu chng ta thay vo s dng $a/m$ lm cutoff, th chng ta nn mong i thy $ma/m = a$ th nghim vt qua cutoff ca chng ta. Do , chng ta tr li vi t l dng tnh gi mong mun l $a$.

Hy th p dng iu ny vo th nghim ca chng ta trn.

Nhng iu Ny Thng Qu Bo Th

V Bon Ferroni rt nghim ngt, bn thng c th b qua nhng mi quan h thc s. C mt v d tt trong bi vit sau:

https://en.wikipedia.org/wiki/Multiple_comparisons_problem

V c bn, n gi nh rng tt c cc bi kim tra bn ang thc hin l c lp, v khng tnh n bt k cu trc no trong d liu ca bn. Bn c th c kh nng thit k mt yu t iu chnh tinh vi hn, nhng iu ny li thm mt lp phc tp v do l mt im tht bi cho nghin cu ca bn. Ni chung, bt c khi no bn ni lng s nghim ngt, bn cn phi rt cn thn khng mc sai lm.

V s qu mc ca Bon Ferroni, thng th vic chy t bi kim tra hn l la chn tt hn. Hoc, nu bn phi chy nhiu bi kim tra, hy dnh ring nhiu b d liu cc tn hiu ng vin ca bn c th tri qua mt vng kim tra ngoi mu. V d, bn c th c quy trnh nh sau:

  • Gi s c 100.000 mi quan h c th.
  • Chy mt bi kim tra cho mi mi quan h c th v chn nhng mi quan h vt qua bi kim tra.
  • Vi nhng ng vin ny, chy mt bi kim tra trn mt b d liu mi ngoi mu. Bi v bn c t ng vin hn, gi y bn c th p dng mt php iu chnh Bon Ferroni cho cc gi tr p ny, hoc nu cn, lp li mt vng kim tra ngoi mu khc.
